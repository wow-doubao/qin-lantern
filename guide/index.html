用法 ​

ts
// main.ts
import { createApp } from 'vue'
import QinLantern from 'qin-lantern'
import 'qin-lantern/dist/style.css'
import App from './App.vue'

const app = createApp(App)

app.use(QinLantern)
app.mount('#app')
